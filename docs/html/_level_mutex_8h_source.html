740<
741 class MutexPolicy,
742 unsigned int DefaultLevel,
743 class ErrorPolicy = ::Loki::ThrowOnBadDesignMutexError,
744 class WaitPolicy = ::Loki::NoMutexWait
745>
746class LevelMutex : public LevelMutexInfo
747{
748public:
749
750 typedef ErrorPolicy EP;
751 typedef WaitPolicy WP;
752 typedef MutexPolicy MP;
753
759 explicit LevelMutex( unsigned int level = DefaultLevel ) :
760 LevelMutexInfo( level ),
761 m_mutex( level )
762 {
763 assert( IsValid() );
764 }
765
767 ~LevelMutex( void )
768 {
769 assert( IsValid() );
770 }
771
778 inline const volatile MutexPolicy & GetMutexPolicy( void ) const volatile { return m_mutex; }
779 inline volatile MutexPolicy & GetMutexPolicy( void ) volatile { return m_mutex; }
780 inline const MutexPolicy & GetMutexPolicy( void ) const { return m_mutex; }
781 inline MutexPolicy & GetMutexPolicy( void ) { return m_mutex; }
782
783 virtual MutexErrors::Type TryLock( void ) volatile
784 {
785 assert( IsValid() );
786 LOKI_MUTEX_DEBUG_CODE( Checker checker( this ); (void)checker; )
787
788 MutexErrors::Type result = LevelMutexInfo::PreLockCheck( true );
789 if ( MutexErrors::Success == result )
790 return MutexErrors::Success;
791 else if ( MutexErrors::AlreadyLocked == result )
792 return result;
793 else if ( MutexErrors::NoProblem != result )
794 return EP::CheckError( result, GetLevel() );
795
796 assert( 0 == LevelMutexInfo::GetLockCount() );
797 result = m_mutex.TryLock();
798 if ( MutexErrors::Success != result )
799 return EP::CheckError( result, GetLevel() );
800 LevelMutexInfo::PostLock();
801
802 return MutexErrors::Success;
803 }
804
805 virtual MutexErrors::Type Lock( void ) volatile
806 {
807 assert( IsValid() );
808 LOKI_MUTEX_DEBUG_CODE( Checker checker( this ); (void)checker; )
809
810 MutexErrors::Type result = LevelMutexInfo::PreLockCheck( false );
811 if ( MutexErrors::Success == result )
812 return MutexErrors::Success;
813 else if ( MutexErrors::NoProblem != result )
814 return EP::CheckError( result, GetLevel() );
815
816 assert( !LevelMutexInfo::IsLockedByCurrentThread() );
817 result = m_mutex.Lock();
818 if ( MutexErrors::Success != result )
819 return EP::CheckError( result, GetLevel() );
820 PostLock();
821
822 return MutexErrors::Success;
823 }
824
825 virtual MutexErrors::Type Lock( unsigned int milliSeconds ) volatile
826 {
827 assert( IsValid() );
828 LOKI_MUTEX_DEBUG_CODE( Checker checker( this ); (void)checker; )
829
830 MutexErrors::Type result = LevelMutexInfo::PreLockCheck( false );
831 if ( MutexErrors::Success == result )
832 return MutexErrors::Success;
833 else if ( MutexErrors::NoProblem != result )
834 return EP::CheckError( result, GetLevel() );
835
836 assert( !LevelMutexInfo::IsLockedByCurrentThread() );
837 clock_t timeOut = clock() + milliSeconds;
838 while ( clock() < timeOut )
839 {
840 WP::Wait();
841 result = m_mutex.TryLock();
842 switch ( result )
843 {
844 case MutexErrors::Success:
845 {
846 PostLock();
847 return MutexErrors::Success;
848 }
849 case MutexErrors::AlreadyLocked:
850 return MutexErrors::AlreadyLocked;
851 case MutexErrors::TryFailed:
852 break;
853 default:
854 return EP::CheckError( result, GetLevel() );
855 }
856 }
857
858 return MutexErrors::TimedOut;
859 }
860
861 virtual MutexErrors::Type Unlock( void ) volatile
862 {
863 assert( IsValid() );
864 LOKI_MUTEX_DEBUG_CODE( Checker checker( this ); (void)checker; )
865
866 MutexErrors::Type result = LevelMutexInfo::PreUnlockCheck();
867 if ( MutexErrors::Success == result )
868 return MutexErrors::Success;
869 else if ( MutexErrors::NoProblem != result )
870 return EP::CheckError( result, GetLevel() );
871
872 LevelMutexInfo::PreUnlock();
873 result = MutexErrors::OtherError;
874 try
875 {
876 result = m_mutex.Unlock();
877 if ( MutexErrors::Success != result )
878 PostLock();
879 }
880 catch ( ... )
881 {
882 PostLock();
883 result = MutexErrors::ExceptionThrown;
884 }
885
886 return result;
887 }
888
889private:
890
892 LevelMutex( const LevelMutex & );
894 LevelMutex & operator = ( const LevelMutex & );
895
896 virtual MutexErrors::Type DoErrorCheck( MutexErrors::Type result ) const volatile
897 {
898 return EP::CheckError( result, GetLevel() );
899 }
900
907 virtual MutexErrors::Type LockThis( void ) volatile
908 {
909 assert( IsValid() );
910 LOKI_MUTEX_DEBUG_CODE( Checker checker( this ); (void)checker; )
911 assert( this != LevelMutexInfo::GetCurrentMutex() );
912
913 const MutexErrors::Type result = m_mutex.Lock();
914 if ( MutexErrors::Success != result )
915 return result;
916 PostLock();
917
918 return MutexErrors::Success;
919 }
920
928 virtual MutexErrors::Type LockThis( unsigned int milliSeconds ) volatile
929 {
930 assert( IsValid() );
931 LOKI_MUTEX_DEBUG_CODE( Checker checker( this ); (void)checker; )
932
933 clock_t timeOut = clock() + milliSeconds;
934 while ( clock() < timeOut )
935 {
936 WP::Wait();
937 const bool locked = ( MutexErrors::Success == m_mutex.TryLock() );
938 if ( locked )
939 {
940 PostLock();
941 return MutexErrors::Success;
942 }
943 }
944
945 return MutexErrors::TimedOut;
946 }
947
954 virtual MutexErrors::Type UnlockThis( void ) volatile
955 {
956 assert( IsValid() );
957 assert( NULL != LevelMutexInfo::GetCurrentMutex() );
958 LOKI_MUTEX_DEBUG_CODE( Checker checker( this ); (void)checker; )
959
960 if ( 1 < LevelMutexInfo::GetLockCount() )
961 {
962 LevelMutexInfo::DecrementCount();
963 return MutexErrors::Success;
964 }
965
966 LevelMutexInfo::PreUnlock();
967 MutexErrors::Type result = m_mutex.Unlock();
968
969 return result;
970 }
