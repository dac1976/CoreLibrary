359 template <typename T, typename A>
360 defs::char_buffer_t const& Build(const T& message, int32_t messageId,
361 const defs::connection_t& responseAddress) const
362 {
363 // Serialise message.
364 serialize::ToCharVector<T, A>(message, m_serialisationBuffer);
365
366 if (m_serialisationBuffer.empty())
367 {
368 BOOST_THROW_EXCEPTION(std::runtime_error("cannot serialize message"));
369 }
370
371 // Resize message buffer.
372 auto totalLength = sizeof(defs::MessageHeader) + m_serialisationBuffer.size();
373 m_messageBuffer.resize(totalLength);
374
375 // Fill header.
376 const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
377 defs::MessageHeader* header =
378 reinterpret_cast<defs::MessageHeader*>(m_messageBuffer.data());
379 FillHeader(m_magicString,
380 archiveType,
381 messageId,
382 responseAddress,
383 static_cast<uint32_t>(m_serialisationBuffer.size()),
384 *header);
385
386 auto writePosIter = std::next(m_messageBuffer.begin(), sizeof(defs::MessageHeader));
387 std::copy(m_serialisationBuffer.begin(), m_serialisationBuffer.end(), writePosIter);
388
389 return m_messageBuffer;
390 }
391
392private:
393#ifdef USE_DEFAULT_CONSTRUCTOR_
395 std::string m_magicString;
396#else
398 std::string m_magicString{static_cast<char const*>(defs::DEFAULT_MAGIC_STRING)};
399#endif
400 mutable defs::char_buffer_t m_messageBuffer;
401 mutable defs::char_buffer_t m_serialisationBuffer;
402};

484template <typename T>
485T DeserializeMessage(const defs::char_buffer_t& messageBuffer, defs::eArchiveType archiveType)
486{
487 assert((archiveType != defs::eArchiveType::raw) &&
488 (archiveType != defs::eArchiveType::protobuf));
489
490 switch (archiveType)
491 {
492 case defs::eArchiveType::binary:
493 return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
494 case defs::eArchiveType::portableBinary:
495 return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
496 case defs::eArchiveType::raw:
497 // Do nothing.
498 break;
499 case defs::eArchiveType::json:
500 return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
501 case defs::eArchiveType::xml:
502 return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
503 case defs::eArchiveType::protobuf:
504 // Do nothing;
505 break;
506 }
507
508 return T();
509}
