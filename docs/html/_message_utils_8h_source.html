268  template <typename T, typename A>
269  defs::char_buffer_t Build(const T& message, int32_t messageId,
270  const defs::connection_t& responseAddress) const
271  {
272  const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
273 
274  auto header = FillHeader(m_magicString, archiveType, messageId, responseAddress);
275  serialize::char_vector_t body = serialize::ToCharVector<T, A>(message);
276 
277  // cppcheck-suppress functionStatic
278  if (body.empty())
279  {
280  BOOST_THROW_EXCEPTION(std::runtime_error("cannot serialize message"));
281  }
282 
283  header.totalLength += static_cast<uint32_t>(body.size());
284 
285  defs::char_buffer_t messageBuffer;
286  messageBuffer.reserve(header.totalLength);
287 
288  auto pHeaderCharBuf = reinterpret_cast<const char*>(&header);
289  std::copy(
290  pHeaderCharBuf, pHeaderCharBuf + sizeof(header), std::back_inserter(messageBuffer));
291 
292  std::copy(body.begin(), body.end(), std::back_inserter(messageBuffer));
293 
294  return messageBuffer;
295  }
296

298 #ifdef USE_DEFAULT_CONSTRUCTOR_
299 
300  std::string m_magicString;
301 #else
302 
303  std::string m_magicString{static_cast<char const*>(defs::DEFAULT_MAGIC_STRING)};
304 #endif

362 template <typename T>
363 T DeserializeMessage(const defs::char_buffer_t& messageBuffer, defs::eArchiveType archiveType)
364 {
365  switch (archiveType)
366  {
367  case defs::eArchiveType::binary:
368  return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
369  case defs::eArchiveType::portableBinary:
370  return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
371  case defs::eArchiveType::raw:
372  return serialize::ToObject<T, serialize::archives::in_raw_t>(messageBuffer);
373  case defs::eArchiveType::json:
374  return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
375  case defs::eArchiveType::xml:
376  return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
377  }

379  return {};
