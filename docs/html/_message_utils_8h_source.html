309  defs::char_buffer_t const&

311  const defs::connection_t& responseAddress,
312  defs::eArchiveType archiveType = defs::eArchiveType::raw) const;
327  template <typename T, typename A>
328  defs::char_buffer_t const& Build(const T& message, int32_t messageId,
329  const defs::connection_t& responseAddress) const
330  {
331  // Serialise message.
332  m_serialisationBuffer = std::move(serialize::ToCharVector<T, A>(message));
333 
334  if (m_serialisationBuffer.empty())
335  {
336  BOOST_THROW_EXCEPTION(std::runtime_error("cannot serialize message"));
337  }
338 
339  // Resize message buffer.
340  auto totalLength = sizeof(defs::MessageHeader) + m_serialisationBuffer.size();
341  m_messageBuffer.resize(totalLength);
342 
343  // Fill header.
344  const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
345  defs::MessageHeader* header =
346  reinterpret_cast<defs::MessageHeader*>(m_messageBuffer.data());
347  FillHeader(m_magicString,
348  archiveType,
349  messageId,
350  responseAddress,
351  static_cast<uint32_t>(m_serialisationBuffer.size()),
352  *header);
353 
354  auto writePosIter = std::next(m_messageBuffer.begin(), sizeof(defs::MessageHeader));
355  std::copy(m_serialisationBuffer.begin(), m_serialisationBuffer.end(), writePosIter);
356 
357  return m_messageBuffer;
358  }
359

361 #ifdef USE_DEFAULT_CONSTRUCTOR_
362

364 #else
365 
366  std::string m_magicString{static_cast<char const*>(defs::DEFAULT_MAGIC_STRING)};
367 #endif
368  mutable defs::char_buffer_t m_messageBuffer;
369  mutable defs::char_buffer_t m_serialisationBuffer;
370 };

452 template <typename T>
453 T DeserializeMessage(const defs::char_buffer_t& messageBuffer, defs::eArchiveType archiveType)
454 {
455  assert((archiveType != defs::eArchiveType::raw) &&
456  (archiveType != defs::eArchiveType::protobuf));
457 
458  switch (archiveType)
459  {
460  case defs::eArchiveType::binary:
461  return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
462  case defs::eArchiveType::portableBinary:
463  return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
464  case defs::eArchiveType::raw:
465  // Do nothing.
466  break;
467  case defs::eArchiveType::json:
468  return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
469  case defs::eArchiveType::xml:
470  return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
471  case defs::eArchiveType::protobuf:
472  // Do nothing;
473  break;
474  }
475 
476  return {};
477 }
