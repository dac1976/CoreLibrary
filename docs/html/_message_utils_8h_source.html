308  template<typename T, typename A>
309  defs::char_buffer_t Build(const T& message
310  , const uint32_t messageId
311  , const defs::connection_t& responseAddress) const
312  {
313  const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
314 
315  auto header = FillHeader(m_magicString, archiveType, messageId, responseAddress);
316  serialize::char_vector_t body
317  = serialize::ToCharVector<T, A>(message);
318 
319  // cppcheck-suppress functionStatic
320  if (body.empty())
321  {
322  BOOST_THROW_EXCEPTION(xArchiveTypeError());
323  }
324 
325  header.totalLength += static_cast<uint32_t>(body.size());
326 
327  defs::char_buffer_t messageBuffer;
328  messageBuffer.reserve(header.totalLength);
329 
330  auto pHeaderCharBuf = reinterpret_cast<const char*>(&header);
331  std::copy(pHeaderCharBuf, pHeaderCharBuf + sizeof(header)
332  , std::back_inserter(messageBuffer));
333 
334  std::copy(body.begin(), body.end()
335  , std::back_inserter(messageBuffer));
336 
337  return messageBuffer;
338  }

341 #ifdef __USE_DEFAULT_CONSTRUCTOR__
342 
343  const std::string m_magicString;
344 #else
345 
346  const std::string m_magicString{ defs::DEFAULT_MAGIC_STRING };

421 template <typename T>
422 T DeserializeMessage(const defs::char_buffer_t& messageBuffer
423  , const defs::eArchiveType archiveType)
424 {
425  switch(archiveType)
426  {
427  case defs::eArchiveType::binary:
428  return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
429  case defs::eArchiveType::portableBinary:
430  return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
431  case defs::eArchiveType::raw:
432  return serialize::ToObject<T, serialize::archives::in_raw_t>(messageBuffer);
433  case defs::eArchiveType::text:
434  return serialize::ToObject<T, serialize::archives::in_txt_t>(messageBuffer);
435  case defs::eArchiveType::xml:
436  return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
437  default:
438  BOOST_THROW_EXCEPTION(xMessageDeserializationError("unsupported archive type"));
439  }
440 
441  return T();
442 }
