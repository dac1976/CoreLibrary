331  defs::char_buffer_t Build(const uint32_t messageId,
332  const defs::connection_t& responseAddress) const;
347  template <typename T, typename A>
348  defs::char_buffer_t Build(const T& message, const uint32_t messageId,
349  const defs::connection_t& responseAddress) const
350  {
351  const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
352 
353  auto header = FillHeader(m_magicString, archiveType, messageId, responseAddress);
354  serialize::char_vector_t body = serialize::ToCharVector<T, A>(message);
355 
356  // cppcheck-suppress functionStatic
357  if (body.empty())
358  {
359  BOOST_THROW_EXCEPTION(xArchiveTypeError());
360  }
361 
362  header.totalLength += static_cast<uint32_t>(body.size());
363 
364  defs::char_buffer_t messageBuffer;
365  messageBuffer.reserve(header.totalLength);
366 
367  auto pHeaderCharBuf = reinterpret_cast<const char*>(&header);
368  std::copy(
369  pHeaderCharBuf, pHeaderCharBuf + sizeof(header), std::back_inserter(messageBuffer));
370 
371  std::copy(body.begin(), body.end(), std::back_inserter(messageBuffer));
372 
373  return messageBuffer;
374  }
375 
376 private:
377 #ifdef USE_DEFAULT_CONSTRUCTOR_
378 
379  const std::string m_magicString;
380 #else
381 
382  const std::string m_magicString{defs::DEFAULT_MAGIC_STRING};
383 #endif
384 };

469 template <typename T>
470 T DeserializeMessage(const defs::char_buffer_t& messageBuffer, const defs::eArchiveType archiveType)
471 {
472  switch (archiveType)
473  {
474  case defs::eArchiveType::binary:
475  return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
476  case defs::eArchiveType::portableBinary:
477  return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
478  case defs::eArchiveType::raw:
479  return serialize::ToObject<T, serialize::archives::in_raw_t>(messageBuffer);
480  case defs::eArchiveType::json:
481  return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
482  case defs::eArchiveType::xml:
483  return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
484  default:
485  BOOST_THROW_EXCEPTION(xMessageDeserializationError("unsupported archive type"));
486  }
487 
488  return T();
489 }
