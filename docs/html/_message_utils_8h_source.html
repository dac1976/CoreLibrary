307  template<typename T, typename A>
308  defs::char_buffer_t Build(const T& message
309  , const uint32_t messageId
310  , const defs::connection_t& responseAddress) const
311  {
312  const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
313 
314  auto header = FillHeader(m_magicString, archiveType, messageId, responseAddress);
315  serialize::char_vector_t body
316  = serialize::ToCharVector<T, A>(message);
317 
318  // cppcheck-suppress functionStatic
319  if (body.empty())
320  {
321  BOOST_THROW_EXCEPTION(xArchiveTypeError());
322  }
323 
324  header.totalLength += static_cast<uint32_t>(body.size());
325 
326  defs::char_buffer_t messageBuffer;
327  messageBuffer.reserve(header.totalLength);
328 
329  auto pHeaderCharBuf = reinterpret_cast<const char*>(&header);
330  std::copy(pHeaderCharBuf, pHeaderCharBuf + sizeof(header)
331  , std::back_inserter(messageBuffer));
332 
333  std::copy(body.begin(), body.end()
334  , std::back_inserter(messageBuffer));
335 
336  return messageBuffer;
337  }
338 
339 private:
340 #ifdef USE_DEFAULT_CONSTRUCTOR_
341 
342  const std::string m_magicString;
343 #else
344 
345  const std::string m_magicString{ defs::DEFAULT_MAGIC_STRING };
346 #endif
347 };

420 template <typename T>
421 T DeserializeMessage(const defs::char_buffer_t& messageBuffer
422  , const defs::eArchiveType archiveType)
423 {
424  switch(archiveType)
425  {
426  case defs::eArchiveType::binary:
427  return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
428  case defs::eArchiveType::portableBinary:
429  return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
430  case defs::eArchiveType::raw:
431  return serialize::ToObject<T, serialize::archives::in_raw_t>(messageBuffer);
432  case defs::eArchiveType::json:
433  return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
434  case defs::eArchiveType::xml:
435  return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
436  default:
437  BOOST_THROW_EXCEPTION(xMessageDeserializationError("unsupported archive type"));
438  }
439 
440  return T();
441 }
