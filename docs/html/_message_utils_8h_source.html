319  defs::char_buffer_t Build(const uint32_t messageId,
320  const defs::connection_t& responseAddress) const;
335  template <typename T, typename A>
336  defs::char_buffer_t Build(const T& message, const uint32_t messageId,
337  const defs::connection_t& responseAddress) const
338  {
339  const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
340 
341  auto header = FillHeader(m_magicString, archiveType, messageId, responseAddress);
342  serialize::char_vector_t body = serialize::ToCharVector<T, A>(message);
343 
344  // cppcheck-suppress functionStatic
345  if (body.empty())
346  {
347  BOOST_THROW_EXCEPTION(xArchiveTypeError());
348  }
349 
350  header.totalLength += static_cast<uint32_t>(body.size());
351 
352  defs::char_buffer_t messageBuffer;
353  messageBuffer.reserve(header.totalLength);
354 
355  auto pHeaderCharBuf = reinterpret_cast<const char*>(&header);
356  std::copy(
357  pHeaderCharBuf, pHeaderCharBuf + sizeof(header), std::back_inserter(messageBuffer));
358 
359  std::copy(body.begin(), body.end(), std::back_inserter(messageBuffer));
360 
361  return messageBuffer;
362  }
363 
364 private:
365 #ifdef USE_DEFAULT_CONSTRUCTOR_
366 
367  const std::string m_magicString;
368 #else
369 
370  const std::string m_magicString{defs::DEFAULT_MAGIC_STRING};
371 #endif
372 };

453 template <typename T>
454 T DeserializeMessage(const defs::char_buffer_t& messageBuffer, const defs::eArchiveType archiveType)
455 {
456  switch (archiveType)
457  {
458  case defs::eArchiveType::binary:
459  return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
460  case defs::eArchiveType::portableBinary:
461  return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
462  case defs::eArchiveType::raw:
463  return serialize::ToObject<T, serialize::archives::in_raw_t>(messageBuffer);
464  case defs::eArchiveType::json:
465  return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
466  case defs::eArchiveType::xml:
467  return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
468  default:
469  BOOST_THROW_EXCEPTION(xMessageDeserializationError("unsupported archive type"));
470  }
471 
472  return T();
473 }
