65template <typename Header>
66class ReceivedMessagePool : public std::enable_shared_from_this<ReceivedMessagePool<Header>>
67{
68 static_assert(std::is_default_constructible<Header>::value,
69 "ReceivedMessagePool requires Header to be default constructible");
70
71public:
72 using msg_t = defs::ReceivedMessage<Header>;
73 using msg_sh_ptr_t = std::shared_ptr<msg_t>;
74
75#ifdef USE_DEFAULT_CONSTRUCTOR_
77 ReceivedMessagePool() {}
78#else
80 ReceivedMessagePool() = default;
81#endif
82
88 ReceivedMessagePool(size_t poolSize, size_t reservedBodySize)
89 : m_pool(poolSize)
90 , m_reservedBodySize(reservedBodySize)
91 {
92 for (size_t i = 0; i < poolSize; ++i)
93 {
94 m_pool[i] = std::make_unique<msg_t>();
95 m_pool[i]->body.reserve(reservedBodySize);
96 m_available.push_back(i);
97 }
98 }
99
101 ~ReceivedMessagePool() = default;
103 ReceivedMessagePool(const ReceivedMessagePool&) = delete;
105 ReceivedMessagePool& operator=(const ReceivedMessagePool&) = delete;
106
111 size_t PoolSize() const
112 {
113 STD_SHARED_LOCK(m_mutex);
114 return m_pool.size();
115 }
116
121 size_t ReservedBodySize() const
122 {
123 return m_reservedBodySize;
124 }
125
134 msg_sh_ptr_t Acquire(size_t requiredBodySize)
135 {
136 auto actualReservedBodySize = ReservedBodySize();
137
138 if (requiredBodySize <= actualReservedBodySize)
139 {
140 STD_UNIQUE_LOCK(m_mutex);
141
142 if (!m_available.empty())
143 {
144 const auto index = m_available.front();
145 m_available.pop_front();
146
147 auto* msg = m_pool[index].get();
148 msg->header = Header{};
149 msg->body.clear();
150
151 auto self = this->shared_from_this();
152
153 return msg_sh_ptr_t(
154 msg,
155 [self, index](msg_t* p)
156 {
157 self->ReleaseToPool(index, p);
158 });
159 }
160 }
161
162 return MakeDynamic(requiredBodySize);
163 }
164
165private:
171 msg_sh_ptr_t MakeDynamic(size_t requiredBodySize)
172 {
173 auto msg = std::make_shared<msg_t>();
174 msg->body.reserve(requiredBodySize);
175 return msg;
176 }
177
183 void ReleaseToPool(size_t index, msg_t* msg)
184 {
185 // Guard against null pointer shenanigans.
186 if (nullptr == msg)
187 {
188 return;
189 }
190
191 msg->header = Header{};
192 msg->body.clear();
193
194 auto actualReservedBodySize = ReservedBodySize();
195
196 // Shrink oversized buffers before reusing.
197 //
198 // This should never get called unless someone tampered with the
199 // message body after acquiring from the pool, but we can be
200 // defensive here anyway, best to be safe.
201 if (msg->body.capacity() > actualReservedBodySize)
202 {
203 defs::char_buffer_t tmp;
204 tmp.reserve(actualReservedBodySize);
205 msg->body.swap(tmp);
206 }
207
208 // Reduce scope of lock to just when we return the index to the pool.
209 STD_UNIQUE_LOCK(m_mutex);
210
211 // Return index to pool of available messages.
212 m_available.push_back(index);
213 }
214
215private:
217 mutable std::shared_mutex m_mutex;
219 using msg_ptr_t = std::unique_ptr<msg_t>;
221 std::vector<msg_ptr_t> m_pool;
223 std::deque<size_t> m_available;
225 size_t m_reservedBodySize{0};
226};

514 Build(defs::char_buf_cspan_t message, int32_t messageId,

516 defs::eArchiveType archiveType = defs::eArchiveType::raw) const;
531 template <typename T, typename A>
532 defs::char_buf_cspan_t Build(const T& message, int32_t messageId,
533 const defs::connection_t& responseAddress) const
534 {
535 // Serialise message.
536 serialize::ToCharVector<T, A>(message, m_serialisationBuffer);
537
538 if (m_serialisationBuffer.empty())
539 {
540 BOOST_THROW_EXCEPTION(std::runtime_error("cannot serialize message"));
541 }
542
543 // Resize message buffer.
544 auto totalLength = sizeof(defs::MessageHeader) + m_serialisationBuffer.size();
545 m_messageBuffer.resize(totalLength);
546
547 // Fill header.
548 const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
549 defs::MessageHeader* header =
550 reinterpret_cast<defs::MessageHeader*>(m_messageBuffer.data());
551 FillHeader(m_magicString,
552 archiveType,
553 messageId,
554 responseAddress,
555 static_cast<uint32_t>(m_serialisationBuffer.size()),
556 *header);
557
558 auto writePosIter = std::next(m_messageBuffer.begin(), sizeof(defs::MessageHeader));
559 std::copy(m_serialisationBuffer.begin(), m_serialisationBuffer.end(), writePosIter);
560
561 return m_messageBuffer;
562 }
563
564private:
565#ifdef USE_DEFAULT_CONSTRUCTOR_
567 std::string m_magicString;
568#else
570 std::string m_magicString{static_cast<char const*>(defs::DEFAULT_MAGIC_STRING)};
571#endif
572 mutable defs::char_buffer_t m_messageBuffer;
573 mutable defs::char_buffer_t m_serialisationBuffer;
574};

659template <typename T>
660T DeserializeMessage(defs::char_buf_cspan_t messageBuffer, defs::eArchiveType archiveType)
661{
662 assert((archiveType != defs::eArchiveType::raw) &&
663 (archiveType != defs::eArchiveType::protobuf) &&
664 (archiveType != defs::eArchiveType::flatBuffer) &&
665 (archiveType != defs::eArchiveType::messagePack));
666
667 switch (archiveType)
668 {
669 case defs::eArchiveType::binary:
670 return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
671 case defs::eArchiveType::portableBinary:
672 return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
673 case defs::eArchiveType::json:
674 return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
675 case defs::eArchiveType::xml:
676 return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
677 case defs::eArchiveType::raw:
678 case defs::eArchiveType::protobuf:
679 case defs::eArchiveType::flatBuffer:
680 case defs::eArchiveType::messagePack:
681 // Do nothing;
682 break;
683
684 }
685
686 return T();
687}
