346 template <typename T, typename A>
347 defs::char_buf_cspan_t Build(const T& message, int32_t messageId,
348 const defs::connection_t& responseAddress) const
349 {
350 // Serialise message.
351 serialize::ToCharVector<T, A>(message, m_serialisationBuffer);
352
353 if (m_serialisationBuffer.empty())
354 {
355 BOOST_THROW_EXCEPTION(std::runtime_error("cannot serialize message"));
356 }
357
358 // Resize message buffer.
359 auto totalLength = sizeof(defs::MessageHeader) + m_serialisationBuffer.size();
360 m_messageBuffer.resize(totalLength);
361
362 // Fill header.
363 const defs::eArchiveType archiveType = ArchiveTypeToEnum<A>().Enumerate();
364 defs::MessageHeader* header =
365 reinterpret_cast<defs::MessageHeader*>(m_messageBuffer.data());
366 FillHeader(m_magicString,
367 archiveType,
368 messageId,
369 responseAddress,
370 static_cast<uint32_t>(m_serialisationBuffer.size()),
371 *header);
372
373 auto writePosIter = std::next(m_messageBuffer.begin(), sizeof(defs::MessageHeader));
374 std::copy(m_serialisationBuffer.begin(), m_serialisationBuffer.end(), writePosIter);
375
376 return m_messageBuffer;
377 }
378
379private:
380#ifdef USE_DEFAULT_CONSTRUCTOR_
382 std::string m_magicString;
383#else
385 std::string m_magicString{static_cast<char const*>(defs::DEFAULT_MAGIC_STRING)};
386#endif
387 mutable defs::char_buffer_t m_messageBuffer;
388 mutable defs::char_buffer_t m_serialisationBuffer;
389};

471template <typename T>
472T DeserializeMessage(defs::char_buf_cspan_t messageBuffer, defs::eArchiveType archiveType)
473{
474 assert((archiveType != defs::eArchiveType::raw) &&
475 (archiveType != defs::eArchiveType::protobuf));
476
477 switch (archiveType)
478 {
479 case defs::eArchiveType::binary:
480 return serialize::ToObject<T, serialize::archives::in_bin_t>(messageBuffer);
481 case defs::eArchiveType::portableBinary:
482 return serialize::ToObject<T, serialize::archives::in_port_bin_t>(messageBuffer);
483 case defs::eArchiveType::raw:
484 // Do nothing.
485 break;
486 case defs::eArchiveType::json:
487 return serialize::ToObject<T, serialize::archives::in_json_t>(messageBuffer);
488 case defs::eArchiveType::xml:
489 return serialize::ToObject<T, serialize::archives::in_xml_t>(messageBuffer);
490 case defs::eArchiveType::protobuf:
491 // Do nothing;
492 break;
493 }
494
495 return T();
496}
