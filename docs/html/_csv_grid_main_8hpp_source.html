346  {

363  string_utils::PackStdString(line);
364 
365  if ((csvfile.tellg() == csvfile.gcount())
366  || csvfile.eof())
367  {

369  {
370  break;
371  }
372  }
373 
374  if (rowCount >= firstRowToLoad)
375  {
376  m_grid.emplace_back(line, options);
377  }
378 
379  ++rowCount;
380  }
381 
382  csvfile.close();
383  }
