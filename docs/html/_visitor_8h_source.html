149// class template Visitor (specialization)
150// This specialization is not present in the book. It makes it easier to define
151// Visitors for multiple types in a shot by using a typelist. Example:
152//
153// class SomeVisitor :
154// public BaseVisitor // required
155// public Visitor<LOKI_TYPELIST_2(RasterBitmap, Paragraph)>
156// {
157// public:
158// void Visit(RasterBitmap&); // visit a RasterBitmap
159// void Visit(Paragraph &); // visit a Paragraph
160// };
162
163 template <class Head, class Tail, typename R>
164 class Visitor<Typelist<Head, Tail>, R, false>
165 : public Visitor<Head, R, false>, public Visitor<Tail, R, false>
166 {
167 public:
168 typedef R ReturnType;
169 // using Visitor<Head, R>::Visit;
170 // using Visitor<Tail, R>::Visit;
171 };

245 <
246 typename R = void,
247 template <typename, class> class CatchAll = DefaultCatchAll,
248 bool ConstVisitable = false
249 >
250 class BaseVisitable;
251
252 template<typename R,template <typename, class> class CatchAll>
253 class BaseVisitable<R, CatchAll, false>
254 {
255 public:
256 typedef R ReturnType;
257 virtual ~BaseVisitable() {}
258 virtual ReturnType Accept(BaseVisitor&) = 0;
259
260 protected: // give access only to the hierarchy
261 template <class T>
262 static ReturnType AcceptImpl(T& visited, BaseVisitor& guest)
263 {
264 // Apply the Acyclic Visitor
265 if (Visitor<T,R>* p = dynamic_cast<Visitor<T,R>*>(&guest))
266 {
267 return p->Visit(visited);
268 }
269 return CatchAll<R, T>::OnUnknownVisitor(visited, guest);
270 }
271 };

273 template<typename R,template <typename, class> class CatchAll>
274 class BaseVisitable<R, CatchAll, true>
275 {
276 public:
277 typedef R ReturnType;
278 virtual ~BaseVisitable() {}
279 virtual ReturnType Accept(BaseVisitor&) const = 0;
280
281 protected: // give access only to the hierarchy
282 template <class T>
283 static ReturnType AcceptImpl(const T& visited, BaseVisitor& guest)

288 return p->Visit(visited);
