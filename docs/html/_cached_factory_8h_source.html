162 class RateLimitedCreation
163 {
164 private:
165 typedef std::vector< clock_t > Vector;
166 Vector m_vTimes;
167 unsigned maxCreation;
168 clock_t timeValidity;
169 clock_t lastUpdate;
170
171 void cleanVector()
172 {
173 using namespace std;
174 clock_t currentTime = clock();
175 D( cout << "currentTime = " << currentTime<< endl; )
176 D( cout << "currentTime - lastUpdate = " << currentTime - lastUpdate<< endl; )
177 if(currentTime - lastUpdate > timeValidity)
178 {
179 m_vTimes.clear();
180 D( cout << " is less than time validity " << timeValidity; )
181 D( cout << " so clearing vector" << endl; )
182 }
183 else
184 {
185 D( cout << "Cleaning time less than " << currentTime - timeValidity << endl; )
186 D( displayVector(); )
187 Vector::iterator newEnd = remove_if(m_vTimes.begin(), m_vTimes.end(), bind2nd(less<clock_t>(), currentTime - timeValidity));
188 // this rearrangement might be costly, consider optimization
189 // by calling cleanVector in less used onCreate function
190 // ... although it may not be correct
191 m_vTimes.erase(newEnd, m_vTimes.end());
192 D( displayVector(); )
193 }
194 lastUpdate = currentTime;
195 }

197 void displayVector()
198 {
199 std::cout << "Vector : ";
200 copy(m_vTimes.begin(), m_vTimes.end(), std::ostream_iterator<clock_t>(std::cout, " "));
201 std::cout << std::endl;
202 }
203#endif
204 protected:
205 RateLimitedCreation() : maxCreation(10), timeValidity(CLOCKS_PER_SEC), lastUpdate(clock())
206 {}

301 // The following class is intented to provide helpers to sort
302 // the container that will hold an eviction score
303 template
304 <
305 typename ST, // Score type
306 typename DT // Data type
307 >
308 class EvictionHelper
309 {
310 protected:
311 typedef typename std::map< DT, ST > HitMap;
312 typedef typename HitMap::iterator HitMapItr;
313 private:
314 typedef std::pair< ST, DT > SwappedPair;
315 typedef std::multimap< ST, DT > SwappedHitMap;
316 typedef typename SwappedHitMap::iterator SwappedHitMapItr;
317 protected:
318 HitMap m_mHitCount;
319
320 // This function sorts the map according to the score
321 // and returns the lower bound of the sorted container
322 DT& getLowerBound(){
323 assert(!m_mHitCount.empty());
324 // inserting the swapped pair into a multimap
325 SwappedHitMap copyMap;
326 for(HitMapItr itr = m_mHitCount.begin(); itr != m_mHitCount.end(); ++itr)
327 copyMap.insert(SwappedPair((*itr).second, (*itr).first));
328 if((*copyMap.rbegin()).first == 0) // the higher score is 0 ...
329 throw EvictionException(); // there is no key evict
330 return (*copyMap.begin()).second;
331 }
332 };

418 // update the counter
419 template<class T> struct updateCounter : public std::unary_function<T, void>
420 {
421 updateCounter(const DT& key): key_(key){}
422 void operator()(T x)
423 {
424 x.second = (x.first == key_ ? (x.second >> 1) | ( 1 << ((sizeof(ST)-1)*8) ) : x.second >> 1);
425 D( std::cout << x.second << std::endl; )
426 }
427 const DT &key_;
428 updateCounter(const updateCounter& rhs) : key_(rhs.key_){}
429 private:
430 updateCounter& operator=(const updateCounter& rhs);
431 };
432 protected:
433 EvictAging(){}
434 virtual ~EvictAging(){}
435
436 // OnStore initialize the counter for the new key
437 // If the key already exists, the counter is reseted
438 void onCreate(const DT& key){
439 EH::m_mHitCount[key] = 0;
440 }
441
442 void onFetch(const DT&){}
443
444 // onRelease increments the hit counter associated with the object
445 // Updating every counters by iterating over the map
446 // If the key is the key of the fetched object :
447 // the counter is shifted to the right and it's MSB is set to 1
448 // else
449 // the counter is shifted to the left
450 void onRelease(const DT& key)
451 {
452 std::for_each(EH::m_mHitCount.begin(), EH::m_mHitCount.end(), updateCounter< typename HitMap::value_type >(key));
453 }

457 EH::m_mHitCount.erase(key);

468 }

514 // Random Eviction policy
515 void evict()
516 {
517 if(m_vKeys.empty())
518 throw EvictionException();
519 size_type random = static_cast<size_type>((m_vKeys.size()*rand())/(static_cast<size_type>(RAND_MAX) + 1));
520 remove(*(m_vKeys.begin()+random));
521 }

705 AbstractProduct* const getPointerToObjectInContainer(ObjVector &entry)
706 {
707 if(entry.empty()) // No object available
708 { // the object will be created in the calling function.
709 // It has to be created in the calling function because of
710 // the variable number of parameters for CreateObject(...) method
711 return NULL;
712 }
713 else

763 // delete the object
764 template<class T> struct deleteObject : public std::unary_function<T, void>
765 {
766 void operator()(T x){ delete x; }
767 };
768
769 // delete the objects in the vector
770 template<class T> struct deleteVectorObjects : public std::unary_function<T, void>
771 {
772 void operator()(T x){
773 ObjVector &vec(x.second);
774 std::for_each(vec.begin(), vec.end(), deleteObject< typename ObjVector::value_type>());
775 }
776 };
777
778 // delete the keys of the map
779 template<class T> struct deleteMapKeys : public std::unary_function<T, void>
780 {
781 void operator()(T x){ delete x.first; }
782 };
783
784 protected:
785 virtual void remove(AbstractProduct * const pProduct)
786 {
787 typename FetchedObjToKeyMap::iterator fetchedItr = providedObjects.find(pProduct);
788 if(fetchedItr!=providedObjects.end()) // object is unreleased.
789 throw CacheException();
790 bool productRemoved = false;
791 typename KeyToObjVectorMap::iterator objVectorItr;
792 typename ObjVector::iterator objItr;
793 for(objVectorItr=fromKeyToObjVector.begin();objVectorItr!=fromKeyToObjVector.end();++objVectorItr)
794 {
795 ObjVector &v((*objVectorItr).second);
796 objItr = remove_if(v.begin(), v.end(), std::bind2nd(std::equal_to<AbstractProduct*>(), pProduct));
797 if(objItr != v.end()) // we found the vector containing pProduct and removed it
798 {
799 onDestroy(pProduct); // warning policies we are about to destroy an object
800 v.erase(objItr, v.end()); // real removing
801 productRemoved = true;
802 break;
803 }
804 }
805 if(productRemoved==false)
806 throw CacheException(); // the product is not in the cache ?!
807 delete pProduct; // deleting it
808 }

810 public:
811 CachedFactory() : factory(), fromKeyToObjVector(), providedObjects(), outObjects(0)
812 {
813 }
814
815 ~CachedFactory()
816 {
817 using namespace std;
818 // debug information
819 SP::onDebug();
820 // cleaning the Cache
821 for_each(fromKeyToObjVector.begin(), fromKeyToObjVector.end(),
822 deleteVectorObjects< typename KeyToObjVectorMap::value_type >()
823 );
824 if(!providedObjects.empty())
825 {
826 // The factory is responsible for the creation and destruction of objects.
827 // If objects are out during the destruction of the Factory : deleting anyway.
828 // This might not be a good idea. But throwing an exception in a destructor is
829 // considered as a bad pratice and asserting might be too much.
830 // What to do ? Leaking memory or corrupting in use pointers ? hmm...
831 D( cout << "====>> Cache destructor : deleting "<< providedObjects.size()<<" in use objects <<====" << endl << endl; )
832 for_each(providedObjects.begin(), providedObjects.end(),
833 deleteMapKeys< typename FetchedObjToKeyMap::value_type >()
834 );
835 }
836 }
