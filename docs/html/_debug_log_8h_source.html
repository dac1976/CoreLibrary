667 std::pair<std::string, std::string> BuildFilePaths(const std::string& logFolderPath,
668 const std::string& logName)
669 {
670 filesys::path p = filesys::system_complete(logFolderPath);
671
672 std::ostringstream ossCurrLogName;
673 ossCurrLogName << logName << m_logExt;
674
675 std::ostringstream ossOldLogName;
676 ossOldLogName << logName << "_old" << m_logExt;
677
678 auto currPath = p;
679 currPath /= ossCurrLogName.str();
680 currPath = filesys::system_complete(currPath);
681
682 auto oldPath = p;
683 oldPath /= ossOldLogName.str();
684 oldPath = filesys::system_complete(oldPath);
685
686 return std::make_pair(currPath.string(), oldPath.string());

730 bool MessageHandler(dl_private::LogQueueMessage& message)
731 {
732 // If message target is file or both then write to file.
733 if (eMsgTarget::console != message.MsgTarget())
734 {
735 if (eLogMessageLevel::copy_log_to_buf == message.ErrorLevel())
736 {
737 CopyLogToBuffer(m_logFilePath, m_oldLogFilePath, m_ofStream);
738 }
739 else if (eLogMessageLevel::copy_mirror_log_to_buf == message.ErrorLevel())
740 {
741 CopyLogToBuffer(m_mirrorLogFilePath, m_oldMirrorLogFilePath, m_ofStreamMirror);
742 }
743 else
744 {
745 CheckLogFileSize(static_cast<long>(message.Message().size()),
746 m_logFilePath,
747 m_oldLogFilePath,
748 m_ofStream);
749
750 CheckLogFileSize(static_cast<long>(message.Message().size()),
751 m_mirrorLogFilePath,
752 m_oldMirrorLogFilePath,
753 m_ofStreamMirror);
754
755 WriteMessageToLog(message, m_ofStream);
756
757 WriteMessageToLog(message, m_ofStreamMirror);
758
759 // Reduce mutex scope.
760 {
761 std::lock_guard<std::mutex> lock{m_mutex};
762
763 m_logStatus = m_ofStream.good();
764 m_mirrorLogStatus = m_ofStreamMirror.good();
765 }
766 }
767 }
768
769 // If message target is not file only then write to console.
770 if (eMsgTarget::file != message.MsgTarget())
771 {
772 WriteToConsole(message);
773 }
774
775 return true;
776 }

831 bool OpenOfStream(const std::string& filePath, eFileOpenOptions fileOptions, std::ofstream& ofs)
832 {
833 if (ofs.is_open())
834 {
835 return ofs.good();
836 }
837
838 auto path = filesys::system_complete(filesys::path(filePath));
839 filesys::create_directories(path.parent_path());
840
841 ofs.open(path.string(),
842 fileOptions == eFileOpenOptions::truncate_file ? std::ofstream::trunc
843 : std::ofstream::app);
844
845 using std::chrono::system_clock;
846 time_t messageTime = system_clock::to_time_t(system_clock::now());
847 std::thread::id noThread;
848 WriteMessageToLog(dl_private::LogQueueMessage("DEBUG LOG STARTED",
849 messageTime,
850 "",
851 "",
852 -1,
853 noThread,
854 eLogMessageLevel::not_defined),
855 ofs);
856
857 if (m_softwareVersion != "")
858 {
859 std::string message("Software Version ");
860 message += m_softwareVersion;
861 WriteMessageToLog(
862 dl_private::LogQueueMessage(
863 message, messageTime, "", "", -1, noThread, eLogMessageLevel::not_defined),
864 ofs);
865 }
866
867 return ofs.good();
868 }

898 void CheckLogFileSize(long requiredSpace, std::string const& logPath,
899 std::string const& oldLogPath, std::ofstream& ofs)
900 {
901 if (!ofs.is_open())
902 {
903 return;
904 }
905
906 auto pos = static_cast<long>(ofs.tellp());
907
908 if ((MaxLogSize() - pos) < requiredSpace)
909 {
910 CloseOfStream(ofs);
911
912#if defined(USE_STD_FILESYSTEM)
913 filesys::copy_file(logPath, oldLogPath, filesys::copy_options::overwrite_existing);
914#else
915#if BOOST_VERSION > 107300
916 filesys::copy_file(logPath, oldLogPath, filesys::copy_options::overwrite_existing);
917#else
918 filesys::copy_file(logPath, oldLogPath, filesys::copy_option::overwrite_if_exists);
919#endif
920#endif
921
922 OpenOfStream(logPath, eFileOpenOptions::truncate_file, ofs);
923 }

1003 void CopyLogToBuffer(std::string const& logPath, std::string const& oldLogPath,
1004 std::ofstream& ofs)
1005 {
1006 if (nullptr == m_logBufPtr)
1007 {
1008 return;
1009 }
1010
1011 auto filePath = m_copyOldLog ? oldLogPath : logPath;
1012
1013 if (!m_copyOldLog)
1014 {
1015 ofs.close();
1016 }
1017
1018 try
1019 {
1020 m_logBufPtr->resize(static_cast<size_t>(filesys::file_size(filePath)));
1021 std::ifstream ifs(filePath, std::ifstream::binary);
1022 ifs.read(m_logBufPtr->data(), m_logBufPtr->size());
1023 }
1024 catch (...)
1025 {
1026 // Do nothing.
1027 }
1028
1029 if (!m_copyOldLog)
1030 {
1031 ofs.open(logPath, std::ofstream::app);
1032 }
1033
1034 m_logCopyEvent.Signal();
1035 }
