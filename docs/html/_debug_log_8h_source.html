650  void OpenOfStream(const std::string& filePath, const eFileOpenOptions fileOptions)
651  {
652  if (m_ofStream.is_open())
653  {
654  return;
655  }
656 
657  m_ofStream.open(filePath,
658  fileOptions == eFileOpenOptions::truncate_file ? std::ofstream::trunc
659  : std::ofstream::app);
660 
661  using std::chrono::system_clock;
662  time_t messageTime = system_clock::to_time_t(system_clock::now());
663  std::thread::id noThread;
664  WriteMessageToLog(dl_private::LogQueueMessage(
665  "DEBUG LOG STARTED", messageTime, "", "", -1, noThread, eLogMessageLevel::not_defined));
666 
667  if (m_softwareVersion != "")
668  {
669  std::string message("Software Version ");
670  message += m_softwareVersion;
671  WriteMessageToLog(dl_private::LogQueueMessage(
672  message, messageTime, "", "", -1, noThread, eLogMessageLevel::not_defined));
673  }
674  }
676  void CloseOfStream()
677  {
678  if (!m_ofStream.is_open())
679  {
680  return;
681  }
682 
683  using std::chrono::system_clock;
684  time_t messageTime = system_clock::to_time_t(system_clock::now());
685  std::thread::id noThread;

688  m_ofStream.close();
689  }
694  void CheckLogFileSize(const long requiredSpace)
695  {
696  if (!m_ofStream.is_open())
697  {
698  return;
699  }
700 
701  long pos = static_cast<long>(m_ofStream.tellp());
702 
703  if ((MaxLogSize() - pos) < requiredSpace)
704  {
705  CloseOfStream();
706  boost::filesystem::copy_file(m_logFilePath,
707  m_oldLogFilePath,
708  boost::filesystem::copy_option::overwrite_if_exists);
709  OpenOfStream(m_logFilePath, eFileOpenOptions::truncate_file);
710  }
711  }
