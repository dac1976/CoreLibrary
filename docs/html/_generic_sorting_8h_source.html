152  {
153  item_compare compare;
154 
155  // we'll iterate over our items and for each item we'll move it to

157  // the other items in the correct direction.
158  for (auto item = begin; item != end; ++item)

173 template <typename T, typename Pred = std::less<T>> class Quick
174 {
175 public:
177  typedef Pred item_compare;
178 
188  template <typename TIterator> static void Sort(TIterator begin, TIterator end)
189  {
190  auto const length = std::distance(begin, end);
191 
192  if (length <= 1)
193  {
194  return;
195  }
196 
197  auto const pivot = std::next(begin, length / 2);
198  item_compare compare;
199  std::nth_element(begin, pivot, end, compare);
200 
201  Sort(begin, pivot);
202  Sort(std::next(pivot), end);
203  }
