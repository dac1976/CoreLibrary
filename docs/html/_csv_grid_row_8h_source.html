405  void OutputRowToStream(std::ostream& os) const
406  {
407  // for each row loop over its columns...
408  size_t col{};
409 
410  for (const auto& cellItem : m_cells)
411  {
412  // let's get our cell value...
413  std::string cell{static_cast<std::string>(cellItem)};
414 
415  // if string contains quotes, insert an
416  // extra quote...
417  size_t pos{cell.find('"')};
418 
419  while (pos < cell.length())
420  {
421  cell.insert(pos, "\"");
422  pos = cell.find('"', pos + 2);
423  }
424 
425  // if cell contains ',', '\n' or '\r' wrap it in quotes...
426  if (cell.find_first_of(",\r\n") != std::string::npos)
427  {
428  cell = "\"" + cell + "\"";
429  }
430 
431  // output corrected cell...
432  os << cell;
433 
434  // add ',' if not last cell on current row...
435  if (col++ < GetSize() - 1)
436  {
437  os << ",";
438  }
439  }

448  void TokenizeLineQuoted(const std::string& line)
449  {
450  typedef boost::tokenizer<boost::escaped_list_separator<char>> Tokenizer;
451  Tokenizer tokzr{line};

453 
454  while (tokIter != tokzr.end())
455  {
456  // cppcheck-suppress eraseDereference
457  std::string tok{*tokIter++};
