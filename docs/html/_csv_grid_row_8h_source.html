389 void OutputRowToStream(std::ostream& os) const
390 {
391 // for each row loop over its columns...
392 size_t col = 0;
393
394 for (const auto& cellItem : m_cells)
395 {
396 // let's get our cell value...
397 std::string cell{static_cast<std::string>(cellItem)};
398
399 // if string contains quotes, insert an
400 // extra quote...
401 size_t pos{cell.find('"')};
402
403 while (pos < cell.length())
404 {
405 cell.insert(pos, "\"");
406 pos = cell.find('"', pos + 2);
407 }
408
409 // if cell contains '"', ',', '\n' or '\r' wrap it in quotes...
410 if (cell.find_first_of("\",\r\n") != std::string::npos)
411 {
412 std::string temp = "\"";
413 temp.append(cell);
414 temp.append("\"");
415 cell.swap(temp);
416 }
417
418 // output corrected cell...
419 os << cell;
420
421 // add ',' if not last cell on current row...
422 if (col++ < GetSize() - 1)
423 {
424 os << ",";
425 }
426 }
427 }

436 void TokenizeLineQuoted(const std::string& line)
437 {
438 using TokenType = boost::escaped_list_separator<char>;
439 using Tokenizer = boost::tokenizer<TokenType>;
440 Tokenizer tokzr(line);
441 auto tokIter = tokzr.begin();
442
443 while (tokIter != tokzr.end())
444 {
445 std::string tok{*tokIter++};
446 boost::trim(tok);
447 m_cells.emplace_back(tok);
448 }
449 }
450
458 void TokenizeLine(const std::string& line)
459 {
460 std::stringstream line_ss{line};
461 std::string tok;
462
463 while (std::getline(line_ss, tok, ','))
464 {
465 tok.shrink_to_fit();
466 boost::trim(tok);
467 m_cells.emplace_back(tok);
468 }
469 }
