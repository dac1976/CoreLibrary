399  void OutputRowToStream(std::ostream& os) const
400  {
401  // for each row loop over its columns...
402  size_t col{};
403 
404  for (const auto& cellItem : m_cells)
405  {
406  // let's get our cell value...
407  std::string cell{static_cast<std::string>(cellItem)};
408 
409  // if string contains quotes, insert an
410  // extra quote...
411  size_t pos{cell.find('"')};
412 
413  while (pos < cell.length())
414  {
415  cell.insert(pos, "\"");
416  pos = cell.find('"', pos + 2);
417  }
418 
419  // if cell contains ',', '\n' or '\r' wrap it in quotes...
420  if (cell.find_first_of(",\r\n") != std::string::npos)
421  {
422  cell = "\"" + cell + "\"";
423  }
424 
425  // output corrected cell...
426  os << cell;
427 
428  // add ',' if not last cell on current row...
429  if (col++ < GetSize() - 1)
430  {
431  os << ",";
432  }
433  }
