414  void LoadFromCSVFile(const std::string& filename, const eCellFormatOptions options,
415  const size_t firstRowToLoad = 0,
416  const size_t maxNumRowsToLoad = std::numeric_limits<size_t>::max())
417  {
418  std::ifstream csvfile{filename.c_str()};
419 
420  if (!csvfile.is_open())
421  {
422  BOOST_THROW_EXCEPTION(xCsvGridCreateFileStreamError(
423  std::string("failed to create file stream for loading: ") + filename));
424  }
425 
426  m_grid.clear();
427  size_t rowCount = 0;
428  std::string row;
429  bool rowComplete = false;
430 
431  while (csvfile.good() && (m_grid.size() < maxNumRowsToLoad))
432  {
433  std::string line;
434  std::getline(csvfile, line);
435  string_utils::PackStdString(line);
436 
437  if ((csvfile.tellg() == csvfile.gcount()) || csvfile.eof())
438  {
439  if (line == "")
440  {
441  break;
442  }
443  }
444 
445  // Because a cell may have line breaks within it we have to
446  // accumulate lines from the file until we have a complete row.
447  // But as we join the lines back to gether to form a complete
448  // row we must add back in the new-line char for each line end.
449  if (!row.empty())
450  {
451  row.append("\n");
452  }
453 
454  row.append(line);
455 
456  if (ContainsEndOfRow(row))
457  {
458  rowComplete = true;
459  }
460 
461  if ((rowCount >= firstRowToLoad) && rowComplete)
462  {
463  m_grid.emplace_back(row, options);
464  }
465 
466  if (rowComplete)
467  {
468  ++rowCount;
469  rowComplete = false;
470  row.clear();
471  }
472  }
473 
474  csvfile.close();
475  }
