356 void LoadFromCSVFile(const std::string& filename, eCellFormatOptions options,
357 size_t firstRowToLoad = 0,
358 size_t maxNumRowsToLoad = std::numeric_limits<size_t>::max())
359 {
360 std::ifstream csvfile{filename.c_str()};
361
362 if (!csvfile.is_open())
363 {
364 std::string err("failed to create ifstream: ");
365 err.append(filename);
366 BOOST_THROW_EXCEPTION(std::runtime_error(err));
367 }
368
369 m_grid.clear();
370 size_t rowCount = 0;
371 std::string row;
372 bool rowComplete = false;
373
374 while (csvfile.good() && (m_grid.size() < maxNumRowsToLoad))
375 {
376 std::string line;
377 std::getline(csvfile, line);
378 line.shrink_to_fit();
379
380 if ((csvfile.tellg() == csvfile.gcount()) || csvfile.eof())
381 {
382 if (line.compare("") == 0)
383 {
384 break;
385 }
386 }
387
388 // Because a cell may have line breaks within it we have to
389 // accumulate lines from the file until we have a complete row.
390 // But as we join the lines back to gether to form a complete
391 // row we must add back in the new-line char for each line end.
392 if (!row.empty())
393 {
394 row.append("\n");
395 }
396
397 row.append(line);
398
399 if (ContainsEndOfRow(row))
400 {
401 rowComplete = true;
402 }
403
404 if ((rowCount >= firstRowToLoad) && rowComplete)
405 {
406 m_grid.emplace_back(row, options);
407 }
408
409 if (rowComplete)
410 {
411 ++rowCount;
412 rowComplete = false;
413 row.clear();
414 }
415 }
416
417 csvfile.close();
418 }
