380  void LoadFromCSVFile(const std::string& filename
381  , const eCellFormatOptions options
382  , const size_t firstRowToLoad = 0
383  , const size_t maxNumRowsToLoad = std::numeric_limits<size_t>::max())
384  {
385  std::ifstream csvfile{filename.c_str()};
386 
387  if (!csvfile.is_open())
388  {
389  BOOST_THROW_EXCEPTION(
390  xCsvGridCreateFileStreamError(std::string("failed to create file stream for loading: ")
391  + filename));
392  }
393 
394  m_grid.clear();
395  size_t rowCount = 0;
396 
397  while(csvfile.good() && (m_grid.size() < maxNumRowsToLoad))
398  {
399  std::string line;
400  std::getline(csvfile, line);
401  string_utils::PackStdString(line);
402 
403  if ((csvfile.tellg() == csvfile.gcount())
404  || csvfile.eof())
405  {
406  if (line == "")
407  {
408  break;
409  }
410  }
411 
412  if (rowCount >= firstRowToLoad)
413  {
414  m_grid.emplace_back(line, options);
415  }
416 
417  ++rowCount;
418  }
419 
420  csvfile.close();
421  }
