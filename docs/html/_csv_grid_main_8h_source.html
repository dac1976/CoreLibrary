408  void LoadFromCSVFile(const std::string& filename
409  , const eCellFormatOptions options
410  , const size_t firstRowToLoad = 0
411  , const size_t maxNumRowsToLoad = std::numeric_limits<size_t>::max())
412  {
413  std::ifstream csvfile{filename.c_str()};
414 
415  if (!csvfile.is_open())
416  {
417  BOOST_THROW_EXCEPTION(
418  xCsvGridCreateFileStreamError(std::string("failed to create file stream for loading: ")
419  + filename));
420  }
421 
422  m_grid.clear();
423  size_t rowCount = 0;
424  std::string row;
425  bool rowComplete = false;
426 
427  while(csvfile.good() && (m_grid.size() < maxNumRowsToLoad))
428  {
429  std::string line;
430  std::getline(csvfile, line);
431  string_utils::PackStdString(line);
432 
433  if ((csvfile.tellg() == csvfile.gcount())
434  || csvfile.eof())
435  {
436  if (line == "")
437  {
438  break;
439  }
440  }
441 
442  // Because a cell may have line breaks within it we have to
443  // accumulate lines from the file until we have a complete row.
444  // But as we join the lines back to gether to form a complete
445  // row we must add back in the new-line char for each line end.
446  if (!row.empty())
447  {
448  row.append("\n");
449  }
450 
451  row.append(line);
452 
453  if (ContainsEndOfRow(row))
454  {
455  rowComplete = true;
456  }
457 
458  if ((rowCount >= firstRowToLoad) && rowComplete)
459  {
460  m_grid.emplace_back(row, options);
461  }
462 
463  if (rowComplete)
464  {
465  ++rowCount;
466  rowComplete = false;
467  row.clear();
468  }
469  }
470 
471  csvfile.close();
472  }
