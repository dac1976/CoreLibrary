407  void LoadFromCSVFile(const std::string& filename
408  , const eCellFormatOptions options
409  , const size_t firstRowToLoad = 0
410  , const size_t maxNumRowsToLoad = std::numeric_limits<size_t>::max())
411  {
412  std::ifstream csvfile{filename.c_str()};
413 
414  if (!csvfile.is_open())
415  {
416  BOOST_THROW_EXCEPTION(
417  xCsvGridCreateFileStreamError(std::string("failed to create file stream for loading: ")
418  + filename));
419  }
420 
421  m_grid.clear();
422  size_t rowCount = 0;
423 
424  while(csvfile.good() && (m_grid.size() < maxNumRowsToLoad))
425  {
426  std::string line;
427  std::getline(csvfile, line);
428  string_utils::PackStdString(line);
429 
430  if ((csvfile.tellg() == csvfile.gcount())
431  || csvfile.eof())
432  {
433  if (line == "")
434  {
435  break;
436  }
437  }
438 
439  if (rowCount >= firstRowToLoad)
440  {
441  m_grid.emplace_back(line, options);
442  }
443 
444  ++rowCount;
445  }
446 
447  csvfile.close();
448  }
