112template < class Value , template<class> class OnError = TriggerAssert >
113class CheckReturn
114{
115public:
116
118 inline CheckReturn( const Value & value ) :
119 m_value( value ), m_checked( false ) {}
120
124 inline CheckReturn( const CheckReturn & that ) :
125 m_value( that.m_value ), m_checked( false )
126 { that.m_checked = true; }
127
129 inline ~CheckReturn( void )
130 {
131 // If m_checked is false, then a function failed to check the
132 // return value from a function call.
133 if (!m_checked)
134 OnError<Value>::run(m_value);
135 }
136
138 inline operator Value ( void )
139 {
140 m_checked = true;
141 return m_value;
142 }
143
144private:
146 CheckReturn( void );
147
149 CheckReturn & operator = ( const CheckReturn & that );
150
152 Value m_value;
153
155 mutable bool m_checked;
156};
