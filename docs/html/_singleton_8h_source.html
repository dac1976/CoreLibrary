147 template <typename T, typename Destroyer>
148 void SetLongevity(T* pDynObject, unsigned int longevity,
149 Destroyer d)
150 {
151 using namespace Private;
152
153 // manage lifetime of stack manually
154 if(pTrackerArray==nullptr)
155 pTrackerArray = new TrackerArray;
156
157 // automatically delete the ConcreteLifetimeTracker object when a exception is thrown
158 // (HGL) - Changed from auto_ptr to unique_ptr.
159 std::unique_ptr<LifetimeTracker>
160 p( new ConcreteLifetimeTracker<T, Destroyer>(pDynObject, longevity, d) );
161
162 // Find correct position
163 TrackerArray::iterator pos = std::upper_bound(
164 pTrackerArray->begin(),
165 pTrackerArray->end(),
166 p.get(),
167 LifetimeTracker::Compare);
168
169 // Insert the pointer to the ConcreteLifetimeTracker object into the queue
170 pTrackerArray->insert(pos, p.get());
171
172 // nothing has thrown: don't delete the ConcreteLifetimeTracker object
173 p.release();
174
175 // Register a call to AtExitFn
176 std::atexit(Private::AtExitFn);
177 }
178
179#else
180
181 template <typename T, typename Destroyer>
182 void SetLongevity(T* pDynObject, unsigned int longevity,
183 Destroyer d)
184 {
185 using namespace Private;
186
187 TrackerArray pNewArray = static_cast<TrackerArray>(
188 std::realloc(pTrackerArray,
189 sizeof(*pTrackerArray) * (elements + 1)));
190 if (!pNewArray) throw std::bad_alloc();
191
192 // Delayed assignment for exception safety
193 pTrackerArray = pNewArray;
194
195 LifetimeTracker* p = new ConcreteLifetimeTracker<T, Destroyer>(
196 pDynObject, longevity, d);
197
198 // Insert a pointer to the object into the queue
199 TrackerArray pos = std::upper_bound(
200 pTrackerArray,
201 pTrackerArray + elements,
202 p,
203 LifetimeTracker::Compare);
204 std::copy_backward(
205 pos,
206 pTrackerArray + elements,
207 pTrackerArray + elements + 1);
208 *pos = p;
209 ++elements;
210
211 // Register a call to AtExitFn
212 std::atexit(Private::AtExitFn);
213 }

215#endif

811 {
