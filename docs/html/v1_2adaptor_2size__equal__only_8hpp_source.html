69template <typename T>
70struct convert<type::size_equal_only<T> > {
71 msgpack::object const& operator()(msgpack::object const& o, type::size_equal_only<T>& v) const {
72 switch(o.type) {
73 case msgpack::type::ARRAY:
74 if (o.via.array.size != msgpack::type::size(v.m_t)) throw msgpack::type_error();
75 break;
76 case msgpack::type::MAP:
77 if (o.via.map.size != msgpack::type::size(v.m_t)) throw msgpack::type_error();
78 break;
79 default:
80 throw msgpack::type_error();
81 }
82 o >> v.m_t;
83 return o;
84 }
85};
