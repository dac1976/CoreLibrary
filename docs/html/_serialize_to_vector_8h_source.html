119template <typename T, typename A> struct ToCharVectorImpl
120{
128 char_vector_t operator()(const T& object) const
129 {
130 std::stringstream os;
131 // Reduce scope of archive to make sure it has
132 // flushed its contents to the stream before
133 // we try and do anything with it.
134 {
135 A oa(os);
136 // CEREAL_NVP / BOOST_SERIALIZATION_NVP is required to fully support xml archives.
137 SERIALIZE_TO_STREAM_ARCHIVE(oa, object);
138 }
139
140 char_vector_t charVector{std::istreambuf_iterator<char>(os),
141 std::istreambuf_iterator<char>()};
142 return charVector;
143 }
144
152 void operator()(const T& object, char_vector_t& result) const
153 {

159 A oa(os);

169template <typename T> struct ToCharVectorImpl<T, archives::out_raw_t>
170{
178 char_vector_t operator()(const T& object) const
179 {
180 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
181
182 auto begin = reinterpret_cast<char const*>(&object);
183 auto end = std::next(begin, static_cast<int>(sizeof(T)));
184 char_vector_t charVector(begin, end);
185 return charVector;
186 }
187
195 void operator()(const T& object, char_vector_t& result) const
196 {
197 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
198
199 auto len = sizeof(T);
200 auto begin = reinterpret_cast<char const*>(&object);
201 auto end = std::next(begin, static_cast<int>(len));
202 result.resize(len);
203 std::copy(begin, end, result.begin());
204 }
205};

258template <typename T, typename A> struct ToObjectImpl
259{
265 T operator()(char_cspan_buf_t charSpan) const
266 {
267 std::stringstream is;
268 std::copy(charSpan.begin(), charSpan.end(), std::ostream_iterator<char>(is));
269 T object;
270 // Reduce scope of archive to make sure it has
271 // flushed its contents to the stream before
272 // we try and do anything with it.
273 {
274 A ia(is);
275 // CEREAL_NVP / BOOST_SERIALIZATION_NVP is required to fully support xml archives.
276 DESERIALIZE_FROM_STREAM_ARCHIVE(ia, object);
277 }
278
279 return object;
280 }
281};

284template <typename T> struct ToObjectImpl<T, archives::in_raw_t>
285{
291 T operator()(char_cspan_buf_t charSpan) const
292 {
293 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
294
295 if (charSpan.size() != sizeof(T))
296 {
297 BOOST_THROW_EXCEPTION(std::invalid_argument("buffer to object size mismatch"));
298 }
299
300 T object{};
301 memcpy(&object, charSpan.data(), charSpan.size());
302 return object;
303 }
304};

307template <typename T> struct ToObjectImpl<T, archives::in_protobuf_t>
308{
314 T operator()(char_cspan_buf_t charSpan) const
315 {
316 std::stringstream is;
317 std::copy(charSpan.begin(), charSpan.end(), std::ostream_iterator<char>(is));
318 T object;
319
320 if (!object.ParseFromIstream(&is))
321 {
322 BOOST_THROW_EXCEPTION(std::runtime_error("failed to deserialize to protocol buffer"));
323 }
324
325 return object;
326 }
