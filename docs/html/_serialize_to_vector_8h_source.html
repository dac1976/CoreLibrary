114template <typename T, typename A> struct ToCharVectorImpl

123 char_vector_t operator()(const T& object) const
124 {
125 std::stringstream os;
126 // Reduce scope of archive to make sure it has
127 // flushed its contents to the stream before
128 // we try and do anything with it.
129 {
130 A oa(os);

164template <typename T> struct ToCharVectorImpl<T, archives::out_raw_t>
165{
173 char_vector_t operator()(const T& object) const
174 {
175 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
176
177 auto begin = reinterpret_cast<char const*>(&object);
178 auto end = std::next(begin, static_cast<int>(sizeof(T)));
179 char_vector_t charVector(begin, end);
180 return charVector;
181 }

252template <typename T, typename A> struct ToObjectImpl
253{
259 T operator()(const char_vector_t& charVector) const
260 {
261 std::stringstream is;
262 std::copy(charVector.begin(), charVector.end(), std::ostream_iterator<char>(is));
263 T object;
264 // Reduce scope of archive to make sure it has
265 // flushed its contents to the stream before
266 // we try and do anything with it.
267 {
268 A ia(is);
269 // CEREAL_NVP / BOOST_SERIALIZATION_NVP is required to fully support xml archives.
270 DESERIALIZE_FROM_STREAM_ARCHIVE(ia, object);
271 }
272
273 return object;
274 }

278template <typename T> struct ToObjectImpl<T, archives::in_raw_t>
279{
285 T operator()(const char_vector_t& charVector) const
286 {
287 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
288
289 if (charVector.size() != sizeof(T))
290 {
291 BOOST_THROW_EXCEPTION(std::invalid_argument("buffer to object size mismatch"));
292 }
293
294 T object{};
295 memcpy(&object, charVector.data(), charVector.size());
296 return object;
297 }
298};

301template <typename T> struct ToObjectImpl<T, archives::in_protobuf_t>
302{
308 T operator()(const char_vector_t& charVector) const
309 {
310 std::stringstream is;
311 std::copy(charVector.begin(), charVector.end(), std::ostream_iterator<char>(is));
312 T object;
313
314 if (!object.ParseFromIstream(&is))
315 {
316 BOOST_THROW_EXCEPTION(std::runtime_error("failed to deserialize to protocol buffer"));
317 }
318
319 return object;
320 }
