94 template <typename T, typename A>
95 struct ToCharVectorImpl
96 {
102  char_vector_t operator()(const T& object) const
103  {
104  std::stringstream os;
105  // Reduce scope of archive to make sure it has
106  // flushed its contents to the stream before
107  // we try and do anything with it.
108  {
109  A oa(os);
110  // CEREAL_NVP is required to fully support xml archives.
111  oa( CEREAL_NVP(object) );
112  }
113  char_vector_t charVector;
114  charVector.reserve(os.str().size());
115  charVector.assign(std::istreambuf_iterator<char>(os)
116  , std::istreambuf_iterator<char>());
117  return charVector;
118  }
119 };
