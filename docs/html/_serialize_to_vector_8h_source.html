112 template <typename T, typename A> struct ToCharVectorImpl
113 {
119  char_vector_t operator()(const T& object) const
120  {
121  std::stringstream os;
122  // Reduce scope of archive to make sure it has
123  // flushed its contents to the stream before
124  // we try and do anything with it.
125  {
126  A oa(os);
127  // CEREAL_NVP is required to fully support xml archives.
128  oa(CEREAL_NVP(object));
129  }
130 
131  char_vector_t charVector;
132  charVector.reserve(os.str().size());
133  charVector.assign(std::istreambuf_iterator<char>(os), std::istreambuf_iterator<char>());
134  return charVector;
135  }

139 template <typename T> struct ToCharVectorImpl<T, archives::out_raw_t>
140 {
146  char_vector_t operator()(const T& object) const
147  {
148  if (!std::is_pod<T>::value)
149  {
150  BOOST_THROW_EXCEPTION(std::invalid_argument("object is not POD"));
151  }
152 
153  char_vector_t charVector;
154  auto begin = reinterpret_cast<char const*>(&object);
155  auto end = std::next(begin, static_cast<int>(sizeof(T)));
156  std::copy(begin, end, std::back_inserter(charVector));
157  return charVector;
158  }
159 };

218 template <typename T> struct ToObjectImpl<T, archives::in_raw_t>
219 {
225  T operator()(const char_vector_t& charVector) const
226  {
227  if (!std::is_pod<T>::value)
228  {
229  BOOST_THROW_EXCEPTION(std::invalid_argument("object is not POD"));
230  }
231 
232  if (charVector.size() != sizeof(T))
233  {
234  BOOST_THROW_EXCEPTION(std::invalid_argument("buffer to object size mismatch"));
235  }
236 
237  T object{};
238  memcpy(&object, charVector.data(), charVector.size());
239  return object;
240  }
