97 template <typename T, typename A> struct ToCharVectorImpl
98 {
104  char_vector_t operator()(const T& object) const
105  {
106  std::stringstream os;
107  // Reduce scope of archive to make sure it has
108  // flushed its contents to the stream before
109  // we try and do anything with it.
110  {
111  A oa(os);
112  // CEREAL_NVP is required to fully support xml archives.
113  oa(CEREAL_NVP(object));
114  }
115 
116  char_vector_t charVector;
117  charVector.reserve(os.str().size());
118  charVector.assign(std::istreambuf_iterator<char>(os), std::istreambuf_iterator<char>());
119  return charVector;
120  }
121 };

124 template <typename T> struct ToCharVectorImpl<T, archives::out_raw_t>
125 {
131  char_vector_t operator()(const T& object) const
132  {
133  char_vector_t charVector;
134 
135  if (!std::is_pod<T>::value)
136  {
137  return charVector;
138  }
139 
140  auto begin = reinterpret_cast<char const*>(&object);
141  auto end = std::next(begin, static_cast<int>(sizeof(T)));
142  std::copy(begin, end, std::back_inserter(charVector));
143  return charVector;
144  }
