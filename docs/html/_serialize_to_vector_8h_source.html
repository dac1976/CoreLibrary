132 char_vector_t operator()(const T& object) const
133 {
134 std::stringstream os;
135 // Reduce scope of archive to make sure it has
136 // flushed its contents to the stream before

139 A oa(os);

173template <typename T> struct ToCharVectorImpl<T, archives::out_raw_t>
174{
182 char_vector_t operator()(const T& object) const
183 {
184 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
185
186 auto begin = reinterpret_cast<char const*>(&object);
187 auto end = std::next(begin, static_cast<int>(sizeof(T)));
188 char_vector_t charVector(begin, end);
189 return charVector;
190 }
191
199 void operator()(const T& object, char_vector_t& result) const
200 {
201 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
202
203 auto len = sizeof(T);
204 auto begin = reinterpret_cast<char const*>(&object);
205 auto end = std::next(begin, static_cast<int>(len));
206 result.resize(len);
207 std::copy(begin, end, result.begin());
208 }
209};

262template <typename T, typename A> struct ToObjectImpl
263{
269 T operator()(char_cspan_buf_t charSpan) const
270 {
271 std::stringstream is;
272 std::copy(charSpan.begin(), charSpan.end(), std::ostream_iterator<char>(is));
273 T object;
274 // Reduce scope of archive to make sure it has
275 // flushed its contents to the stream before
276 // we try and do anything with it.
277 {
278 A ia(is);
279 // CEREAL_NVP / BOOST_SERIALIZATION_NVP is required to fully support xml archives.
280 DESERIALIZE_FROM_STREAM_ARCHIVE(ia, object);
281 }
282
283 return object;
284 }
285};

288template <typename T> struct ToObjectImpl<T, archives::in_raw_t>
289{
295 T operator()(char_cspan_buf_t charSpan) const
296 {
297 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
298
299 if (charSpan.size() != sizeof(T))
300 {
301 BOOST_THROW_EXCEPTION(std::invalid_argument("buffer to object size mismatch"));
302 }
303
304 T object{};
305 memcpy(&object, charSpan.data(), charSpan.size());
306 return object;
307 }
308};

311template <typename T> struct ToObjectImpl<T, archives::in_protobuf_t>
312{
318 T operator()(char_cspan_buf_t charSpan) const
319 {
320 std::stringstream is;
321 std::copy(charSpan.begin(), charSpan.end(), std::ostream_iterator<char>(is));
322 T object;
323
324 if (!object.ParseFromIstream(&is))
325 {
326 BOOST_THROW_EXCEPTION(std::runtime_error("failed to deserialize to protocol buffer"));
327 }
328
329 return object;
330 }

547template <typename T, typename VerifyFunc, typename GetRootFunc>
548T ToObjectFlatBuf(char_cspan_buf_t charSpan, VerifyFunc verifyFunc, GetRootFunc getRootFunc)
549{
550 const auto* p = reinterpret_cast<const uint8_t*>(charSpan.data());
551 const auto n = charSpan.size();
552
553 flatbuffers::Verifier v(p, n);
554
555 if (!verifyFunc(v))
556 {
557 BOOST_THROW_EXCEPTION(std::runtime_error("failed to verify flatbuffer data"));
558 }
559
560 // Most generated getters take ONLY the pointer.
561 // If your getRootFunc expects const void*, this still works.
562 auto* view = getRootFunc(p);
563
564 T obj;
565 view->UnPackTo(&obj);
566 return obj; // NRVO/move
567}
