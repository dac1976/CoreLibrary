138 template <typename T, typename A>
139 struct ToCharVectorImpl
140 {
148 char_vector_t operator()(const T &object) const
149 {
150 std::stringstream os;
151 // Reduce scope of archive to make sure it has
152 // flushed its contents to the stream before
153 // we try and do anything with it.
154 {
155 A oa(os);
156 // CEREAL_NVP / BOOST_SERIALIZATION_NVP is required to fully support xml archives.
157 SERIALIZE_TO_STREAM_ARCHIVE(oa, object);
158 }
159
160 char_vector_t charVector{std::istreambuf_iterator<char>(os),
161 std::istreambuf_iterator<char>()};
162 return charVector;
163 }

189 template <typename T>
190 struct ToCharVectorImpl<T, archives::out_raw_t>
191 {
199 char_vector_t operator()(const T &object) const
200 {
201 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
202
203 auto begin = reinterpret_cast<char const *>(&object);
204 auto end = std::next(begin, static_cast<int>(sizeof(T)));
205 char_vector_t charVector(begin, end);
206 return charVector;
207 }
208
216 void operator()(const T &object, char_vector_t &result) const
217 {
218 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
219
220 auto len = sizeof(T);
221 auto begin = reinterpret_cast<char const *>(&object);
222 auto end = std::next(begin, static_cast<int>(len));
223 result.resize(len);
224 std::copy(begin, end, result.begin());
225 }

315 template <typename T, typename A>
316 struct ToObjectImpl
317 {
323 T operator()(char_cspan_buf_t charSpan) const
324 {
325 std::stringstream is;
326 std::copy(charSpan.begin(), charSpan.end(), std::ostream_iterator<char>(is));
327 T object;
328 // Reduce scope of archive to make sure it has
329 // flushed its contents to the stream before
330 // we try and do anything with it.
331 {
332 A ia(is);
333 // CEREAL_NVP / BOOST_SERIALIZATION_NVP is required to fully support xml archives.
334 DESERIALIZE_FROM_STREAM_ARCHIVE(ia, object);
335 }
336
337 return object;
338 }

342 template <typename T>
343 struct ToObjectImpl<T, archives::in_raw_t>
344 {
350 T operator()(char_cspan_buf_t charSpan) const
351 {
352 static_assert(std::is_trivially_copyable<T>::value, "object should be POD");
353
354 if (charSpan.size() != sizeof(T))
355 {
356 BOOST_THROW_EXCEPTION(std::invalid_argument("buffer to object size mismatch"));
357 }
358
359 T object{};
360 memcpy(&object, charSpan.data(), charSpan.size());
361 return object;
362 }

390 template <typename T>
391 struct ToObjectImpl<T, archives::in_msgpack_t>
392 {
398 T operator()(char_cspan_buf_t charSpan) const
399 {
400 if (charSpan.empty())
401 {
402 BOOST_THROW_EXCEPTION(std::runtime_error("Cannot deserialize MessagePack object from empty buffer."));
403 }
404
405 msgpack::object_handle oh = msgpack::unpack(charSpan.data(), charSpan.size());
406
407 T object;
408 oh.get().convert(object);
409
410 return object;

610 template <typename T, typename VerifyFunc, typename GetRootFunc>
611 T ToObjectFlatBuf(char_cspan_buf_t charSpan, VerifyFunc verifyFunc, GetRootFunc getRootFunc)
612 {
613 const auto *p = reinterpret_cast<const uint8_t *>(charSpan.data());
614 const auto n = charSpan.size();
615
616 flatbuffers::Verifier v(p, n);
617
618 if (!verifyFunc(v))
619 {
620 BOOST_THROW_EXCEPTION(std::runtime_error("failed to verify flatbuffer data"));
621 }
622
623 // Most generated getters take ONLY the pointer.
624 // If your getRootFunc expects const void*, this still works.
625 auto *view = getRootFunc(p);
626
627 T obj;
628 view->UnPackTo(&obj);
629 return obj; // NRVO/move
630 }
