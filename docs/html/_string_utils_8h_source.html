80template <typename T>
81std::string FormatFloatString(const T value, const int precision = 15,
82 eFloatStringFormat formatting = eFloatStringFormat::normal)
83{
84 std::ostringstream ss;
85
86 switch (formatting)
87 {
88 case eFloatStringFormat::fixed:
89 ss << std::fixed;
90 break;
91 case eFloatStringFormat::scientific:
92 ss << std::scientific;
93 break;
94 case eFloatStringFormat::normal:
95 // do nothing
96 break;
97 }
98
99 if (precision >= 0)
100 {
101 ss << std::setprecision(precision);
102 }
103
104 ss << value;
105 return ss.str();
106}

117template <typename T> std::string AutoFormatFloatString(const T value, int decimalPlaces = 1)
118{
119 std::string formattedValue;
120
121 if (decimalPlaces < 0)
122 {
123 decimalPlaces = 0;
124 }
125
126 const T min = std::pow(T(10), T(-1) * decimalPlaces);
127 const T absVal = std::fabs(value);
128
129 if (absVal < std::numeric_limits<T>::epsilon()) // 0.0 case
130 {
131 formattedValue = FormatFloatString(value, 1 + decimalPlaces);
132 }
133 else if ((absVal < min) || (absVal >= T(100000))) // very small and very big values
134 {
135 formattedValue = FormatFloatString(value, decimalPlaces, eFloatStringFormat::scientific);
136 }
137 else // everything in between
138 {
139 if (absVal < T(1))
140 {
141 formattedValue = FormatFloatString(value, decimalPlaces);
142 }
143 else if (absVal < T(10))
144 {
145 formattedValue = FormatFloatString(value, 1 + decimalPlaces);
146 }
147 else if (absVal < T(100))
148 {
149 formattedValue = FormatFloatString(value, 2 + decimalPlaces);
150 }
151 else if (absVal < T(1000))
152 {
153 formattedValue = FormatFloatString(value, 3 + decimalPlaces);
154 }
155 else if (absVal < T(10000))
156 {
157 formattedValue = FormatFloatString(value, 4 + decimalPlaces);
158 }
159 else
160 {
161 formattedValue = FormatFloatString(value, 5 + decimalPlaces);
162 }
163 }
164
165 return formattedValue;
166}

269template <typename Iter>
270void MakeHexStream(std::ostream& os, Iter first, Iter last, bool useUppercase, bool insertSeparator,
271 char separator = ' ')
272{
273 os << std::hex << std::setfill('0');
274
275 if (useUppercase)
276 {
277 os << std::uppercase;
278 }
279
280 while (first != last)
281 {
282 os << std::setw(2) << static_cast<int>(*first++);
283
284 if (insertSeparator && first != last)
285 {
286 os << separator;
287 }
288 }
289}

311template <typename T>
312void MakeHexStream(std::ostream& os, T value, bool useUppercase = true,
313 bool insertLeadingSymbol = true, size_t minNumHexDigits = 0)
314{
315 static_assert(std::is_unsigned<T>::value, "T must be an unsigned integral type");
316
317 os << std::hex << std::setfill('0');
318
319 if (insertLeadingSymbol)
320 {
321 os << "0x";
322 }
323
324 if (useUppercase)
325 {
326 os << std::uppercase;
327 }
328
329 if (0 == minNumHexDigits)
330 {
331 os << std::setw(sizeof(value) * 2);
332 }
333 else
334 {
335 os << std::setw(minNumHexDigits);
336 }
337
338 os << value;
339}

569template <typename TInt>
570std::pair<TInt, bool> SafeConvertCharArrayToInt(const char* text, size_t length)
571{
572 static_assert(std::is_integral<TInt>::value, "TInt must be an integral type");
573
574 std::pair<TInt, bool> result(0, false);
575
576 if ((nullptr == text) || (0 == length))
577 {
578 return result;
579 }
580
581 // Adjust length to the first NULL within [text, text + length), if any.
582 if (const void* p = std::memchr(text, '\0', length))
583 {
584 length = static_cast<const char*>(p) - text;
585 }
586
587 if (0 == length)
588 {
589 return result;
590 }
591
592#if defined(HGL_CPP17) && !defined(NO_FROM_CHARS)
593 auto [ptr, res] = std::from_chars(text, text + length, result.first);
594 result.second = res == std::errc();
595#else
596 try
597 {
598 // We'll furst try to convert to a signed long long int (i.e.in64_t)
599 // then we'll static case to the proper type from there.
600 result.first = static_cast<TInt>(std::stoll(std::string(text, length)));
601 result.second = true;
602 }
603 catch (...)
604 {
605 // Do nothing.
606 }
607#endif
608
609 return result;
610}

680template <typename... Args>
681std::enable_if_t<(std::is_same_v<std::string_view, Args> && ...), std::string>
682FastConcatenateStrings(Args... args)
683{
684 // Calculate the total length of all substrings
685 size_t totalLength = (args.size() + ... + 0);
686
687 // Reserve the correct amount of bytes
688 std::string result;
689 result.reserve(totalLength);
690
691 // Append all substrings
692 (result += ... += args);
693
694 return result;
695}
