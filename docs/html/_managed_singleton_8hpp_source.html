78public:
91 template <typename... Args>
92 static T& Instance(Args&&... args)
93 {
94 T* ptr = m_instance.load(std::memory_order_acquire);
95
96 if (nullptr == ptr)
97 {
98 std::lock_guard<std::mutex> lock(m_mutex);
99
100 ptr = m_instance.load(std::memory_order_relaxed);
101
102 if (nullptr == ptr)
103 {
104 std::unique_ptr<T> newInstance =
105 std::make_unique<T>(std::forward<Args>(args)...);
106
107 ptr = newInstance.get();
108 m_owned = std::move(newInstance);
109 m_instance.store(ptr, std::memory_order_release);
110 }
111 }
112
113 return *ptr;
114 }

144 }
