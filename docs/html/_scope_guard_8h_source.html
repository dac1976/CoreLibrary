60 template <typename J>
61 static void SafeExecute(J& j) throw()
62 {
63 if (!j.dismissed_)
64 try
65 {
66 j.Execute();
67 }
68 catch(...)
69 {}
70 }
71
72 mutable bool dismissed_;
73
74 public:
75 ScopeGuardImplBase() throw() : dismissed_(false)
76 {}
77
78 void Dismiss() const throw()
79 {
80 dismissed_ = true;
81 }
