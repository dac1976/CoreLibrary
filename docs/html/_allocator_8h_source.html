101 size_type max_size( void ) const throw()
102 {
103 // A good optimizer will see these calculations always produce the same
104 // value and optimize this function away completely.
105 const size_type max_bytes = size_type( -1 );
106 const size_type bytes = max_bytes / sizeof( Type );
107 return bytes;
108 }

118 void destroy( pointer p )

120 // If the Type has no destructor, then some compilers complain about
121 // an unreferenced parameter, so use the void cast trick to prevent
122 // spurious warnings.
123 (void)p;
124 p->~Type();
125 }
